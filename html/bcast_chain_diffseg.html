adapt_loop = []
table = []

with open('bcast_chain_diffseg.csv','r') as csvfile:
    plots = csv.reader(csvfile, delimiter=',')
    for row in plots:
        print(row)
        table.append(row);
        if plots.line_num > 1:
            x.append(int(row[0]))
            tuned.append(float(row[1]))
            adapt_2K.append(float(row[2]))
            adapt_4K.append(float(row[3]))

            adapt_16K.append(float(row[5]))

            adapt_256K.append(float(row[9]))
            adapt_512K.append(float(row[10]))
            adapt_curvefit.append(float(row[11]))
            adapt_loop.append(float(row[12]))
